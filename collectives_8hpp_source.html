18  if (realHandle->probed.first) { // previously probed, return 0 if EOS received
19  size=realHandle->probed.second;

24  errno = EBADF; // the "communicator" is not valid or closed
25  return -1;
26  }
27  if (realHandle->closed_rd) return 0;
28 
29  // reading the header to get the size of the message
30  ssize_t r;
31  if ((r=realHandle->probe(size, blocking))<=0) {
32  switch(r) {
33  case 0: {
34  realHandle->close(true, true);
35  return 0;
36  }
37  case -1: {
38  if (errno==ECONNRESET) {
39  realHandle->close(true, true);
40  return 0;
41  }
42  if (errno==EWOULDBLOCK || errno==EAGAIN) {
43  errno = EWOULDBLOCK;
44  return -1;
45  }
46  }}
47  return r;
48  }
49  realHandle->probed={true,size};
50  if (size==0) { // EOS received
51  realHandle->close(false, true);
52  return 0;
53  }
54  return r;
55  }

58  size_t sz;
59  if (!realHandle->probed.first) {
60  // reading the header to get the size of the message
61  ssize_t r;
62  if ((r=probeHandle(realHandle, sz, true))<=0) {
63  return r;
64  }
65  } else {
66  if (!realHandle) {
67  MTCL_PRINT(100, "[internal]:\t", "HandleUser::probe EBADF\n");
68  errno = EBADF; // the "communicator" is not valid or closed
69  return -1;
70  }
71  if (realHandle->closed_rd) return 0;
72  }
73  if ((sz=realHandle->probed.second)>size) {
74  MTCL_ERROR("[internal]:\t", "HandleUser::receive ENOMEM, receiving less data\n");
75  errno=ENOMEM;
76  return -1;
77  }
78  realHandle->probed={false,0};
79  return realHandle->receive(buff, std::min(sz,size));
